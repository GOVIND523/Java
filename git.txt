# git is a open source version control system 

# It helps to maintain a recoird of changes made to a directory or roll back changes to a certain commit.

# git has a gui interface and a bash line terminal 

# git "commands":

       $git init          -  to initialize a empty git repository
       $ls -a             -  hidden files begins with (.)
       $ls .git           -  open a repository
       $touch names.txt   -  creates a names.txt files
       $git status        -  shows the staus of the git repository
       $git add .         -  staging all the files to commit and save changes
       $git commmit -m "Give your message here" 
       $git restore --staged names.txt  -  unstage the staged files
       $git log           -  record of all commit
       $git remote -v     -  Github link
       $git tag           -  version number
       $git log --pretty=oneline
       $git log --online --decorate --graph --all
